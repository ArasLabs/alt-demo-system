var defaultSearchInfo = "";

function changeSearchInfo(searchType, isDefault, div) {
	var searchInfoDiv = document.getElementById("SearchInfo");
	var searchInfo;
	
	switch (searchType) {
		case "total":
			searchInfo = "Search the Web for engineering pages";
			break;
		case "appnotes":
			searchInfo = "Search the Web for application notes";
			break;
		case "patents":
			searchInfo = "Search the Web for patent information";
			break;
		case "standards":
			searchInfo = "Search the Web for engineering standards";
			break;
		case "materials":
			searchInfo = "Search the Web for information on material properties";
			break;
		case "partnumbers":
			searchInfo = "Search for parts by part number, manufacturer name, distributor name, or description";
			break;
		case "products":
			searchInfo = "Search online vendor, manufacturer and supplier product catalogs";
			break;
		case "services":
			searchInfo = "Search online service provider and consultant catalogs";
			break;
		case "prodannounce":
			searchInfo = "Search supplier product announcements on GlobalSpec";
			break;
		case "thissite": //added for 'this site' searched - RH 3/20/2006
			searchInfo = "Search  for engineering pages";
			break;
		case "engnews":
			searchInfo = "Search Engineering news stories from hundreds of sources";
			break;
		default:
			searchInfo = "";
	}
	
	searchInfoDiv.innerHTML = searchInfo;
	
	if (isDefault) {
		defaultSearchInfo = searchInfo;
		
		if (div != null) {
			div.style.backgroundColor = "";
		}
	} else if (div != null) {
		div.style.backgroundColor = "#F7F7F7";
	}
}

function resetSearchInfo(div) {
	var searchInfoDiv = document.getElementById("SearchInfo");
	searchInfoDiv.innerHTML = defaultSearchInfo;

	if (div != null) {
		div.style.backgroundColor = "";
	}
}



function changeSearchLocation(searchType, searchQuery, fromSearchBox) {

	
		var engWebURL = "http://search.globalspec.com/Search?query=";
		var productFinderURL = "http://www.globalspec.com/ProductFinder/FindProducts?query=";
		var productAnnouncementsURL = "http://www.globalspec.com/FeaturedProducts/SearchResults?criteria=";

		if (fromSearchBox) {
			var engnewsUrl = "http://www.globalspec.com/engnews/search?q=";
		} else {
			var engnewsUrl = "http://www.globalspec.com/engnews/?q=";
		}
	
		var gotoURL;
		
		// Depending on search type, we will be going to different URLs for the results
		switch (searchType) {
			case "total":
			case "companysites":
			case "appnotes":
			case "patents":
			case "standards":
			case "materials":
			case "partnumbers":
			case "thissite":	//added for 'this site' searches - RH 3/20/2006
				gotoURL = engWebURL + encodeURIComponent(searchQuery) + "&show=" + searchType;
				break;
			case "products":
				gotoURL = productFinderURL + encodeURIComponent(searchQuery);				
				break;
			case "services":				
				gotoURL = productFinderURL + encodeURIComponent(searchQuery) + "&mode=services";				
				break;
			case "prodannounce":
				gotoURL = productAnnouncementsURL + encodeURIComponent(searchQuery);
				break;
			case "engnews":
				gotoURL = engnewsUrl + encodeURIComponent(searchQuery);
				break;
			default:
				gotoURL = engWebURL + encodeURIComponent (searchQuery)+ "&show=total";
				break;
		}
		
		//alert(gotoURL);
		location.href = gotoURL;
}
